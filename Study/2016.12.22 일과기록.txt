** 별탑 출력하기(하나의  for문으로)

public class PrintTriangle {

    public String printTriangle(int num){
    	
    	String result = "";
        String stars = "*";
        
        for(int i=0; i<num; ++i){
            result += stars+"\n";
            stars += "*";
        }
        
        return result;
    }
}
